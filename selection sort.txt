#include <iostream>
#include <algorithm>
using namespace std;

int main() {
    int a[] = {2, 5, 3, 7, 6};

    cout << "unsorted: ";
    for (int x : a) cout << x << " ";
    cout << "\n";

    sort(begin(a), end(a));   // STL sort

    cout << "sorted: ";
    for (int x : a) cout << x << " ";
    cout << "\n";
}
