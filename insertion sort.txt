#include <iostream>
using namespace std;
int main() {
    int a[] = {13,12,11,9,7},n=5;
    
    for(int i=1;i<n;i++) {
        int k=a[i],j=i-1;
        while(j>=0 && a[j]>k) {
            a[j+1] = a[j];
            j--;
        }
            a[j+1]=k;
    }
            for(int x:a) cout<<x<<" ";
}
